47  inline DacDmaWriterFeature<TDacAlignmentFeature,TPriority,TFifoMode>::DacDmaWriterFeature(Dma& dma)
48  : DmaFeatureBase(dma) {
49 
50  /*
51  * The target for the DMA transfer depends on the alignment, channel (or dual channel).
52  * All of these are available as compile-time constants in the alignment feature therefore
53  * the optimiser will eliminate all the conditional tests below, remove the dead code and emit
54  * only the resulting single storage instructions.
55  */
56 
57  if(TDacAlignmentFeature::DUAL_CHANNEL_FEATURE) {
58 
59  // both channels
60 
61  if(TDacAlignmentFeature::getAlignment==DAC_Align_12b_R)
62  _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR12RD));
63  else if(TDacAlignmentFeature::getAlignment==DAC_Align_12b_L)
64  _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR12LD));
65  else if(TDacAlignmentFeature::getAlignment==DAC_Align_8b_R)
66  _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR8RD));
67  }
68  else {
69 
70  if(TDacAlignmentFeature::getChannel()==DAC_Channel_1) {
71 
72  // channel 1
73 
74  if(TDacAlignmentFeature::getAlignment==DAC_Align_12b_R)
75  _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR12R1));
76  else if(TDacAlignmentFeature::getAlignment==DAC_Align_12b_L)
77  _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR12L1));
78  else if(TDacAlignmentFeature::getAlignment==DAC_Align_8b_R)
79  _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR8R1));
80  }
81  else {
82 
83  // channel 2
84 
85  if(TDacAlignmentFeature::getAlignment==DAC_Align_12b_R)
86  _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR12R2));
87  else if(TDacAlignmentFeature::getAlignment==DAC_Align_12b_L)
88  _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR12L2));
89  else if(TDacAlignmentFeature::getAlignment==DAC_Align_8b_R)
90  _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR8R2));
91  }
92  }
93 
94  if(TDacAlignmentFeature::DUAL_CHANNEL_FEATURE) {
95 
96  // 8 bit = half-word (packed channels: 22222221111111) transfers
97 
98  if(TDacAlignmentFeature::getAlignment()==DAC_Align_8b_R) {
99  _init.DMA_PeripheralDataSize=DMA_PeripheralDataSize_HalfWord;
100  _init.DMA_MemoryDataSize=DMA_MemoryDataSize_HalfWord;
101  }
102  else {
103 
104  // 12 bit = word (packed channels: xxxx222222222222xxxx111111111111) transfers
105 
106  _init.DMA_PeripheralDataSize=DMA_PeripheralDataSize_Word;
107  _init.DMA_MemoryDataSize=DMA_MemoryDataSize_Word;
108  }
109 
110  }
111  else {
112 
113  // 8 bit = byte transfers
114 
115  if(TDacAlignmentFeature::getAlignment()==DAC_Align_8b_R) {
116  _init.DMA_PeripheralDataSize=DMA_PeripheralDataSize_Byte; // transferring bytes
117  _init.DMA_MemoryDataSize=DMA_MemoryDataSize_Byte; // transferring bytes
118  }
119  else {
120 
121  // 12 bit = half-word transfers
122 
123  _init.DMA_PeripheralDataSize=DMA_PeripheralDataSize_HalfWord; // transferring half-words
124  _init.DMA_MemoryDataSize=DMA_MemoryDataSize_HalfWord; // transferring half-words
125  }
126  }

128  _init.DMA_Channel=dma.getChannelNumber(); // channel id
129  _init.DMA_DIR=DMA_DIR_MemoryToPeripheral; // 'peripheral' is destination
130  _init.DMA_PeripheralInc=DMA_PeripheralInc_Disable; // 'peripheral' does not increment
131  _init.DMA_MemoryInc=DMA_MemoryInc_Enable; // memory is incremented

139  _init.DMA_FIFOThreshold=DMA_FIFOThreshold_HalfFull; // flush on half-full
140 
141  DAC_DMACmd(TDacAlignmentFeature::getChannel(),ENABLE);
142  }
