69  template<Orientation TOrientation,ColourDepth TColourDepth,class TAccessMode,class TPanelTraits>
70  inline void SSD1963<TOrientation,TColourDepth,TAccessMode,TPanelTraits>::initialise() const {
71 
72  const unsigned char FSMC_Bank = 0;
73 
74  // slow down the device for the register setup
75 
76  FSMC_Bank1->BTCR[FSMC_Bank+1] = FSMC_BTR1_ADDSET_0 | FSMC_BTR1_DATAST_2 | FSMC_BTR1_BUSTURN_0 ;
77  FSMC_Bank1->BTCR[FSMC_Bank] = FSMC_BCR1_MWID_0 | FSMC_BCR1_WREN | FSMC_BCR1_MBKEN;
78 
79  // do a software reset
80 
81  this->_accessMode.writeCommand(ssd1963::SOFTWARE_RESET);
82  MillisecondTimer::delay(1);
83 
84  // reset the device
85 
86  this->_accessMode.reset();
87 
88  // set the PLL multipler. the pixel clock is sourced from the PLL.
89  // PLL frequency = (external oscillator frequency * N) / M
90  // where 250Mhz < (external oscillator frequency * N) < 800Mhz
91 
92  this->_accessMode.writeCommand(ssd1963::SET_PLL_MN);
93  this->_accessMode.writeData(TPanelTraits::PLL_M);
94  this->_accessMode.writeData(TPanelTraits::PLL_N);
95  this->_accessMode.writeData(4);
96 
97  // enable the PLL
98 
99  this->_accessMode.writeCommand(ssd1963::SET_PLL);
100  this->_accessMode.writeData(1);
101  MillisecondTimer::delay(2);
102  this->_accessMode.writeCommand(ssd1963::SET_PLL);
103  this->_accessMode.writeData(3);
104  MillisecondTimer::delay(2);
105 
106  // do a software reset
107 
108  this->_accessMode.writeCommand(ssd1963::SOFTWARE_RESET);
109  MillisecondTimer::delay(1);
110 
111  // set the pixel clock
112 
113  this->_accessMode.writeCommand(ssd1963::SET_LSHIFT_FREQUENCY);
114  this->_accessMode.writeData(TPanelTraits::PIXEL_CLOCK >> 16);
115  this->_accessMode.writeData((TPanelTraits::PIXEL_CLOCK >> 8) & 0xff);
116  this->_accessMode.writeData(TPanelTraits::PIXEL_CLOCK & 0xff);
117 
118  // set up the LCD mode (colour depth, dithering)
119 
120  this->_accessMode.writeCommand(ssd1963::SET_LCD_MODE);
121  this->_accessMode.writeData(this->getLcdModeBits()); // 0=18 bit, 1=24 bit
122  this->_accessMode.writeData(0); // TFT mode
123  this->_accessMode.writeData((TPanelTraits::LONG_SIDE-1) >> 8); // horizontal size
124  this->_accessMode.writeData((TPanelTraits::LONG_SIDE-1) & 0xff);
125  this->_accessMode.writeData((TPanelTraits::SHORT_SIDE-1) >> 8); // vertical size
126  this->_accessMode.writeData((TPanelTraits::SHORT_SIDE-1) & 0xff);
127  this->_accessMode.writeData(0); // RGB transfer order
128 
129  // horizontal timing
130 
131  this->_accessMode.writeCommand(ssd1963::SET_HORIZONTAL_PERIOD);
132  this->_accessMode.writeData((TPanelTraits::H_BACK_PORCH+TPanelTraits::H_PERIOD+TPanelTraits::H_FRONT_PORCH) >> 8);
133  this->_accessMode.writeData((TPanelTraits::H_BACK_PORCH+TPanelTraits::H_PERIOD+TPanelTraits::H_FRONT_PORCH) & 0xff);
134  this->_accessMode.writeData(TPanelTraits::H_BACK_PORCH >> 8);
135  this->_accessMode.writeData(TPanelTraits::H_BACK_PORCH & 0xff);
136  this->_accessMode.writeData(TPanelTraits::H_SYNC_PULSE_WIDTH-1);
137  this->_accessMode.writeData(TPanelTraits::H_SYNC_POSITION >> 8);
138  this->_accessMode.writeData(TPanelTraits::H_SYNC_POSITION & 0xff);
139  this->_accessMode.writeData(0);
140 
141  // vertical timing
142 
143  this->_accessMode.writeCommand(ssd1963::SET_VERTICAL_PERIOD);
144  this->_accessMode.writeData((TPanelTraits::V_BACK_PORCH+TPanelTraits::V_PERIOD+TPanelTraits::V_FRONT_PORCH) >> 8);
145  this->_accessMode.writeData((TPanelTraits::H_BACK_PORCH+TPanelTraits::V_PERIOD+TPanelTraits::V_FRONT_PORCH) & 0xff);
146  this->_accessMode.writeData(TPanelTraits::V_BACK_PORCH >> 8);
147  this->_accessMode.writeData(TPanelTraits::V_BACK_PORCH & 0xff);
148  this->_accessMode.writeData(TPanelTraits::V_SYNC_PULSE_WIDTH-1);
149  this->_accessMode.writeData(TPanelTraits::V_SYNC_POSITION >> 8);
150  this->_accessMode.writeData(TPanelTraits::V_SYNC_POSITION & 0xff);
151  this->_accessMode.writeData(0);
152 
153  // enable packed 12-bit mode (2 16-bit FSMC transfers per pixel). This requires
154  // that DB0..DB11 are connected. Any other pins exposed by the interface
155  // (i.e. DB12..23) are unused.
156 
157  this->_accessMode.writeCommand(ssd1963::SET_PIXEL_DATA_INTERFACE);
158  this->_accessMode.writeData(1);
159  MillisecondTimer::delay(5);
160 
161  // start up the display
162 
163  this->_accessMode.writeCommand(ssd1963::SLEEP_OUT);
164  MillisecondTimer::delay(10);
165  this->_accessMode.writeCommand(ssd1963::DISPLAY_INVERSION_OFF);
166  this->_accessMode.writeCommand(ssd1963::IDLE_MODE_OFF);
167  this->_accessMode.writeCommand(ssd1963::NORMAL_DISPLAY_MODE_ON);
168 
169  // set the panel orientation
170 
171  this->_accessMode.writeCommand(ssd1963::SET_ADDRESS_MODE,this->getAddressMode());
172 
173  // reset the scroll position
174 
175  this->setScrollPosition(0);
176 
177  // wait the required number of ms. before we can turn the display on
178 
179  MillisecondTimer::delay(125);
180  this->_accessMode.writeCommand(ssd1963::DISPLAY_ON);
181 
182  // the user needs to speed up the FSMC now that this has completed
183  }
