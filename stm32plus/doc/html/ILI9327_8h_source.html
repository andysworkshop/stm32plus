62 
63  // reset the device
64 
65  this->_accessMode.reset();
66 
67  // power setting: we use the internal ref vci=2.5v
68 
69  this->_accessMode.writeCommand(ili9327::PowerSettingCmd::Opcode,ili9327::PowerSettingCmd::VC_100); // vci1=vci*1 (vci=2.5V therefore vci1=2.5V)
70  this->_accessMode.writeData(ili9327::PowerSettingCmd::BT_5_4); // VGH=vci1*5, VGL=vci1*-4
71  this->_accessMode.writeData(ili9327::PowerSettingCmd::VRH(0xc) | ili9327::PowerSettingCmd::VCIRE); // int ref + VREG1OUT = vci1*1.9 = 4.75
72 
73  // VCom (more power settings)
74 
75  this->_accessMode.writeCommand(ili9327::VComCmd::Opcode,0); // register D1 for setting VCom
76  this->_accessMode.writeData(ili9327::VComCmd::VCM(0x40)); // VCOMH = VREG1OUT * 0.748
77  this->_accessMode.writeData(ili9327::VComCmd::VDV(0xf)); // VCOM amplitude=VREG1OUT*1
78 
79  // power setting for normal mode
80 
81  this->_accessMode.writeCommand(ili9327::PowerSettingNormalModeCmd::Opcode,ili9327::PowerSettingNormalModeCmd::AP(1)); // drivers on
82  this->_accessMode.writeData(ili9327::PowerSettingNormalModeCmd::DC(4) | ili9327::PowerSettingNormalModeCmd::DC10(4)); // fosc ratios
83 
84  // exit sleep mode
85 
86  this->_accessMode.writeCommand(ili9327::ExitSleepModeCmd::Opcode);
87  MillisecondTimer::delay(50);
88 
89  // enter normal mode
90 
91  this->_accessMode.writeCommand(ili9327::EnterNormalModeCmd::Opcode);
92 
93  // panel driving setting
94 
95  this->_accessMode.writeCommand(ili9327::PanelDrivingCmd::Opcode,0);
96  this->_accessMode.writeData(ili9327::PanelDrivingCmd::NL(0x35));
97  this->_accessMode.writeData(ili9327::PanelDrivingCmd::SCN(0));
98  this->_accessMode.writeData(ili9327::PanelDrivingCmd::PTS(0)); // modify for power reduction in back/front porch
99  this->_accessMode.writeData(ili9327::PanelDrivingCmd::ISC(1) | ili9327::PanelDrivingCmd::PTG);
100  this->_accessMode.writeData(ili9327::PanelDrivingCmd::DIVE(0));
101 
102  // display timing (c1)
103 
104  this->_accessMode.writeCommand(ili9327::DisplayTimingSettingNormalModeCmd::Opcode,ili9327::DisplayTimingSettingNormalModeCmd::BC | ili9327::DisplayTimingSettingNormalModeCmd::DIV0_1); // line inversion, 1:1 internal clock
105  this->_accessMode.writeData(ili9327::DisplayTimingSettingNormalModeCmd::RTN(0x10)); // 1 line = 16 clocks
106  this->_accessMode.writeData(ili9327::DisplayTimingSettingNormalModeCmd::BP(2));
107  this->_accessMode.writeData(ili9327::DisplayTimingSettingNormalModeCmd::FP(4));
108 
109  // display timing idle (c3)
110 
111  this->_accessMode.writeCommand(ili9327::DisplayTimingSettingIdleModeCmd::Opcode,ili9327::DisplayTimingSettingIdleModeCmd::BC2);
112  this->_accessMode.writeData(ili9327::DisplayTimingSettingIdleModeCmd::RTN2(0x10));
113  this->_accessMode.writeData(ili9327::DisplayTimingSettingIdleModeCmd::BP2(0x2));
114  this->_accessMode.writeData(ili9327::DisplayTimingSettingIdleModeCmd::FP2(0x4));
115 
116  // frame rate = 96Hz
117 
118  this->_accessMode.writeCommand(ili9327::FrameRateAndInversionControlCmd::Opcode,ili9327::FrameRateAndInversionControlCmd::FR_96);
119 
120  // interface control
121 
122  this->_accessMode.writeCommand(ili9327::InterfaceControlCmd::Opcode,ili9327::InterfaceControlCmd::DPL | ili9327::InterfaceControlCmd::EPL | ili9327::InterfaceControlCmd::HSPL | ili9327::InterfaceControlCmd::VSPL);
123 
124  // frame memory access (set DFM for 2 transfers/1 pixel in 18-bit mode)
125 
126  this->_accessMode.writeCommand(ili9327::FrameMemoryAccessAndInterfaceSettingCmd::Opcode,0);
127  this->_accessMode.writeData(0);
128  this->_accessMode.writeData(0);
129  this->_accessMode.writeData(ili9327::FrameMemoryAccessAndInterfaceSettingCmd::DFM);
130 
131  // display on - cannot happen <120ms after reset
132 
133  MillisecondTimer::delay(100);
134  this->_accessMode.writeCommand(ili9327::SetDisplayOnCmd::Opcode);
135 
136  // set the colour mode (base class specialisation does this)
137 
138  this->setColourMode();
139 
140  // set the address mode (orientation from specialisation and BGR order)
141 
142  this->_accessMode.writeCommand(ili9327::
143  SetAddressModeCmd::Opcode,
144  this->getOrientationAddressMode() | ili9327::SetAddressModeCmd::BGR | ili9327::SetAddressModeCmd::VERTICAL_FLIP);
145 
146  // set scroll area to omit the missing scan lines if panel is shorter than 432
147 
148  this->_accessMode.writeCommand(ili9327::SetScrollAreaCmd::Opcode,(432-TPanelTraits::LONG_SIDE) >> 8);
149  this->_accessMode.writeData((432-TPanelTraits::LONG_SIDE) & 0xff);
150  this->_accessMode.writeData(TPanelTraits::LONG_SIDE >> 8);
151  this->_accessMode.writeData(TPanelTraits::LONG_SIDE & 0xff);
152  this->_accessMode.writeData(0);
153  this->_accessMode.writeData(0);
