67  template<class TPeripheralInfo,uint16_t TTimerEvent,uint32_t TPriority,uint32_t TDmaMode>
68  inline void TimerDmaFeature<TPeripheralInfo,TTimerEvent,TPriority,TDmaMode>::beginWriteByTimer(Timer& timer,const void *source,uint32_t count) {
69 
70  DMA_Channel_TypeDef *peripheralAddress;
71 
72  // set up the parameters for this transfer
73 
74  _init.DMA_MemoryBaseAddr=reinterpret_cast<uint32_t>(source);
75  _init.DMA_BufferSize=count;
76 
77  // set the peripheral address from the overloaded operator
78 
79  peripheralAddress=_dma;
80 
81  // connect the timer to its associated DMA channel
82 
83  TIM_DMACmd(timer,TTimerEvent,ENABLE);
84 
85  // disable and then re-enable DMA
86 
87  DMA_Cmd(peripheralAddress,DISABLE);
88  DMA_Init(peripheralAddress,&_init);
89  DMA_Cmd(peripheralAddress,ENABLE);
90  }
