164  FSMC_ReadWriteTimingStruct=const_cast<Fsmc8080LcdTiming *> (&timing);

166  FSMC_AsynchronousWait=FSMC_AsynchronousWait_Disable;

168  // initialise the FSMC and then enable it
169 
170  FSMC_NORSRAMInit(this);

172  // calculate the data address
