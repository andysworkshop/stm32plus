158  inline uint32_t RtcBase::getTick() const {
159 
160  RTC_DateTypeDef rdate;
161  RTC_TimeTypeDef rtime;
162  struct tm tstruct;
163 
164  // get time and date
165 
166  RTC_GetTime(RTC_Format_BIN,&rtime);
167  RTC_GetDate(RTC_Format_BIN,&rdate);
168 
169  // if in 24 hour clock or hour is less than 12 then believe the hours
170 
171  if(_hourFormat==RTC_HourFormat_24 || rtime.RTC_H12==RTC_H12_AM)
172  tstruct.tm_hour=rtime.RTC_Hours;
173  else
174  tstruct.tm_hour=12+rtime.RTC_Hours; // adjust hours + 12 for PM
175 
176  tstruct.tm_min=rtime.RTC_Minutes;
177  tstruct.tm_sec=rtime.RTC_Seconds;
178 
179  tstruct.tm_year=rdate.RTC_Year+100; // years since 1900. 00+100=2000
180  tstruct.tm_mon=rdate.RTC_Month-1;
181  tstruct.tm_mday=rdate.RTC_Date;
182 
183  return mktime(&tstruct);
184  }

192  inline void RtcBase::setTick(uint32_t tick) const {
193 
194  struct tm *tstruct;
195 
196  tstruct=gmtime(reinterpret_cast<time_t *>(&tick));
197 
198  // date is straightforward
199 
200  setDate(tstruct->tm_year-100,tstruct->tm_mon+1,tstruct->tm_mday,tstruct->tm_wday==0 ? 7 : tstruct->tm_wday);
201 
202  // if in 24 hours then easy else the hour must be mod 12 and am/pm set
203 
204  if(_hourFormat==RTC_HourFormat_24)
205  setTime(tstruct->tm_hour,tstruct->tm_min,tstruct->tm_sec,0);
206  else
207  setTime(tstruct->tm_hour % 12,tstruct->tm_min,tstruct->tm_sec,tstruct->tm_hour>=12 ? RTC_H12_PM : RTC_H12_AM);
208  }
