120  inline void GraphicTerminal<TGraphicsLibrary,THardwareScrolling>::calcTerminalSize() {
121 
122  const FontChar *fc;
123 
124  // need to know the width of the characters. They're all the same so measure a space.
125 
126  _font->getCharacter(static_cast<uint8_t>(' '),fc);
127 
128  _fontSize.Height=_font->getHeight();
129  _fontSize.Width=fc->PixelWidth;
130 
131  // height is rounded down if the fixed lines don't sum to a multiple of the font height
132 
133  _terminalSize.Width=_gl.getWidth()/fc->PixelWidth;
134  _terminalSize.Height=_gl.getHeight()/_font->getHeight();
135  }

198  inline void GraphicTerminal<TGraphicsLibrary,THardwareScrolling>::writeCharacter(char c) {
199 
200  char buffer[2];
201  Point p;
202 
203  if(c=='\n') {
204 
205  incrementY();
206  _cursor.X=0;
207 
208  } else if(c=='\r') {
209 
210  if(_autoLineFeed)
211  incrementY();
212 
213  _cursor.X=0;
214  } else {
215 
216  // scale up the x,y character co-ords to pixel co-ords
217 
218  p.X=_cursor.X*_fontSize.Width;
219  p.Y=(_cursor.Y % _terminalSize.Height)*_fontSize.Height;
220 
221  // create a string
222 
223  buffer[0]=c;
224  buffer[1]='\0';
225 
226  // move the cursor and write
227 
228  _gl.writeString(p,*_font,buffer);
229 
230  if(++_cursor.X >= _terminalSize.Width) {
231  _cursor.X=0;
232  incrementY();
233  }
234  }
235  }

262  template<class TGraphicsLibrary,bool THardwareScrolling>
263  void GraphicTerminal<TGraphicsLibrary,THardwareScrolling>::scroll() {
264 
265  int i;
266  Rectangle rc;
267  int16_t screenHeight;
268 
269  screenHeight=_gl.getHeight();
270 
271  rc.Width=_gl.getWidth();
272  rc.X=0;
273 
274  // smooth scroll
275 
276  rc.Height=1;
277 
278  for(i=0;i<_fontSize.Height;i++) {
279 
280  _scrollPosition++;
281 
282  if(_scrollPosition==screenHeight)
283  _scrollPosition=0;
284 
285  _gl.setScrollPosition(_scrollPosition);
286 
287  rc.Y=_scrollPosition>0 ? _scrollPosition-1 : screenHeight-1;
288  _gl.clearRectangle(rc);
289 
290  MillisecondTimer::delay(_smoothScrollStep);
291  }
292 
293  // don't allow the cursor to run away and overflow
294 
295  if(_cursor.Y==_terminalSize.Height*2)
296  _cursor.Y=_terminalSize.Height;
297  }
