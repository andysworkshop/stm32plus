36  // Constructor. Defaults to intializing with nullptr.
37  // There is no way to create an uninitialized scoped_ptr.
38  // The input parameter must be allocated with new.
39  explicit scoped_ptr(C* p=nullptr) :
40  ptr_(p) {
41  }
42 
43  // Destructor. If there is a C object, delete it.
44  // We don't need to test ptr_ == nullptr because C++ does that for us.
45  ~scoped_ptr() {
46  enum {
47  type_must_be_complete=sizeof(C)
48  };
49  delete ptr_;
50  }

52  // Reset. Deletes the current owned object, if any.
53  // Then takes ownership of a new object, if given.
54  // this->reset(this->get()) works.
55  void reset(C* p=nullptr) {
56  if(p != ptr_) {
57  enum {
58  type_must_be_complete=sizeof(C)
59  };
60  delete ptr_;
61  ptr_=p;
62  }
63  }
64 
65  // Accessors to get the owned object.
66  // operator* and operator-> will assert() if there is no current object.
67  C& operator*() const {
68  assert(ptr_ != nullptr);
69  return *ptr_;
70  }
71  C* operator->() const {
72  assert(ptr_ != nullptr);
73  return ptr_;
74  }
75  C* get() const {
76  return ptr_;
77  }
78 
79  // Comparison operators.
80  // These return whether two scoped_ptr refer to the same object, not just to
81  // two different but equal objects.
82  bool operator==(C* p) const {
83  return ptr_ == p;
84  }
85  bool operator!=(C* p) const {
86  return ptr_ != p;
87  }

89  // Swap two scoped pointers.
90  void swap(scoped_ptr& p2) {
91  C* tmp=ptr_;
92  ptr_=p2.ptr_;
93  p2.ptr_=tmp;
94  }

96  // Release a pointer.
97  // The return value is the current pointer held by this object.
98  // If this object holds a nullptr pointer, the return value is nullptr.
99  // After this operation, this object will hold a nullptr pointer,
100  // and will not own the object any more.
101  C* release() {
102  C* retVal=ptr_;
103  ptr_=nullptr;
104  return retVal;
105  }
