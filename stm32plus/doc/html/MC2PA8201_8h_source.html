67  template<Orientation TOrientation,ColourDepth TColourDepth,class TAccessMode,class TPanelTraits>
68  inline void MC2PA8201<TOrientation,TColourDepth,TAccessMode,TPanelTraits>::initialise() const {
69 
70  // reset the device
71 
72  this->_accessMode.reset();
73 
74  // start up the display
75 
76  this->_accessMode.writeCommand(mc2pa8201::SLEEP_OUT);
77  MillisecondTimer::delay(10);
78  this->_accessMode.writeCommand(mc2pa8201::DISPLAY_INVERSION_OFF);
79  this->_accessMode.writeCommand(mc2pa8201::IDLE_MODE_OFF);
80  this->_accessMode.writeCommand(mc2pa8201::NORMAL_DISPLAY_MODE_ON);
81 
82  // interface pixel format comes from the colour specialisation
83 
84  this->_accessMode.writeCommand(mc2pa8201::INTERFACE_PIXEL_FORMAT,this->getInterfacePixelFormat());
85 
86  // memory access control comes from the orientation specialisation
87 
88  this->_accessMode.writeCommand(mc2pa8201::MEMORY_ACCESS_CONTROL,this->getMemoryAccessControl());
89 
90  // reset the scrolling area
91 
92  setScrollArea(0,320);
93 
94  // wait the required number of ms. before we can turn the display on
95 
96  MillisecondTimer::delay(125);
97  this->_accessMode.writeCommand(mc2pa8201::DISPLAY_ON);
98  }
