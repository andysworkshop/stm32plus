183  inline void CharacterLcdTerminal<TImpl>::incrementY() {
184 
185  int16_t x,y;
186  char *ptr;
187 
188  // easy if not yet at the bottom of the display
189 
190  if(_cursor.Y!=_terminalSize.Height-1)
191  _cursor.Y++;
192  else {
193 
194  // now we 'scroll' the framebuffer and clear out the last line
195 
196  memmove(_framebuffer,_framebuffer+_terminalSize.Width,_terminalSize.Width*(_terminalSize.Height-1));
197  memset(_framebuffer+_terminalSize.Width*(_terminalSize.Height-1),' ',_terminalSize.Width);
198 
199  // refresh the whole frame
200 
201  ptr=_framebuffer;
202  for(y=0;y<_terminalSize.Height;y++) {
203  _lcd.moveCursor(0,y);
204  for(x=0;x<_terminalSize.Width;x++) {
205  _lcd.write(*ptr++);
206  }
207  }
208  }
