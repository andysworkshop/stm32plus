66  template<Orientation TOrientation,ColourDepth TColourDepth,class TAccessMode>
67  inline void SSD1289<TOrientation,TColourDepth,TAccessMode>::initialise() const {
68 
69  // reset the device
70 
71  this->_accessMode.reset();
72 
73  // init sequence: datashee 15.5 "Display ON Sequence"
74 
75  // GON = 1, DTE = 0, D[1:0] = 01
76  this->_accessMode.writeCommand(ssd1289::DISPLAY_CONTROL,0x21);
77 
78  // oscillator ON
79  this->_accessMode.writeCommand(ssd1289::OSCILLATOR,0x01);
80 
81  // GON = 1, DTE = 0, D[1:0] = 11
82  this->_accessMode.writeCommand(ssd1289::DISPLAY_CONTROL,0x23);
83 
84  // Exit sleep mode
85  this->_accessMode.writeCommand(ssd1289::SLEEP_MODE,0x00);
86 
87  MillisecondTimer::delay(35);
88 
89  // GON = 1, DTE = 1, D[1:0] = 11
90  this->_accessMode.writeCommand(ssd1289::DISPLAY_CONTROL,0x33);
91 
92  // BGR etc.
93  this->_accessMode.writeCommand(ssd1289::DRIVER_OUTPUT_CONTROL,0x13f | 0x2000 | 0x800 | 0x200);
94 
95  // entry mode: colour and orientation
96  this->_accessMode.writeCommand(ssd1289::ENTRY_MODE,this->getColourEntryMode() | this->getOrientationEntryMode());
97 
98  // driving waveform control
99  this->_accessMode.writeCommand(ssd1289::DRIVING_WAVEFORM_CONTROL,0x600);
100  }
