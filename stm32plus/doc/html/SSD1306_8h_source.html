109  inline void SSD1306<TOrientation,TAccessMode>::initialise() const {
110 
111  // the reset sequence must happen after Vdd has become stable. We'll give it 10ms.
112 
113  MillisecondTimer::delay(10);
114 
115  // reset the device
116 
117  this->_accessMode.reset();
118 
119  // initialise the panel
120 
121  setDisplayOnOff(0); // Display Off (0x00/0x01)
122  setDisplayClock(0x80); // Set Clock as 100 Frames/Sec
123  setMultiplexRatio(0x3F); // 1/64 Duty (0x0F~0x3F)
124  setDisplayOffset(0); // Shift Mapping RAM Counter (0x00~0x3F)
125  setStartLine(0); // Set Mapping RAM Display Start Line (0x00~0x3F)
126  setChargePump(0x04); // Enable Embedded DC/DC Converter (0x00/0x04)
127  setCommonRemap(0x8); // Set COM/Row Scan Direction (0x00/0x08)
128  setCommonConfig(0x10); // Set Sequential Configuration (0x00/0x10)
129  setAddressingMode(0);
130 
131  // orientation-specific stuff
132 
133  this->setOrientation();
134 
135  // continue
136 
137  setContrast(_contrast); // Set SEG Output Current
138  setPrechargePeriod(0xF1); // Set Pre-Charge as 15 Clocks & Discharge as 1 Clock
139  setVComh(0x40); // Set VCOM Deselect Level
140  setEntireDisplay(0); // Disable Entire Display On (0x00/0x01)
141  setInverseDisplay(0); // Disable Inverse Display On (0x00/0x01)
142 
143  // turn the display on
144 
145  setDisplayOnOff(1); // Display On (0x00/0x01)
146  }

386  inline void SSD1306<TOrientation,TAccessMode>::writePixel(const UnpackedColour& cr) {
387 
388  uint8_t value;
389  uint16_t byteIndex;
390  uint32_t bitIndex;
391 
392  byteIndex=((this->_cursorPos.Y/8)*128)+this->_cursorPos.X;
393  bitIndex=(byteIndex*8)+(this->_cursorPos.Y % 8);
394 
395  if(cr.value)
396  _frameBuffer.getBitbandAddress()[bitIndex]=1;
397  else
398  _frameBuffer.getBitbandAddress()[bitIndex]=0;
399 
400  value=_frameBuffer[byteIndex];
401  this->_accessMode.writeData(&value,1);
402 
403  // update cursor position
404 
405  if(++this->_cursorPos.X==this->_window.X+this->_window.Width) {
406  this->_cursorPos.X=this->_window.X;
407  this->_cursorPos.Y++;
408  }

419  template<Orientation TOrientation,class TAccessMode>
420  inline void SSD1306<TOrientation,TAccessMode>::fillPixels(uint32_t numPixels,const UnpackedColour& cr) {
421 
422  uint8_t value,bitpos;
423  uint8_t *ptr;
424  uint16_t index,valuesToWrite,lastIndex;
425 
426  // optimisation for clearing the display
427 
428  if(numPixels==128*64) {
429  memset(&_frameBuffer[0],cr.value ? 0xff : 0,1024);
430  this->_accessMode.writeData(&_frameBuffer[0],1024);
431  }
432  else {
433 
434  // do each pixel separately
435 
436  ptr=nullptr;
437 
438  while(numPixels--) {
439 
440  index=((this->_cursorPos.Y/8)*128)+this->_cursorPos.X;
441  bitpos=this->_cursorPos.Y % 8;
442 
443  value=_frameBuffer[index];
444  value=(value & _plotMasks[bitpos]) | (cr.value << bitpos);
445 
446  _frameBuffer[index]=value;
447 
448  if(ptr==nullptr) {
449  ptr=&_frameBuffer[index];
450  valuesToWrite=1;
451  lastIndex=index;
452  }
453  else {
454  if(lastIndex!=index) {
455  lastIndex=index;
456  valuesToWrite++;
457  }
458  }
459 
460  // update cursor position
461 
462  if(++this->_cursorPos.X==this->_window.X+this->_window.Width) {
463  this->_cursorPos.X=this->_window.X;
464  this->_cursorPos.Y++;
465  }
466  }
467 
468  // write the buffer (optimise this to just write the affected section)
469 
470  this->_accessMode.writeData(ptr,valuesToWrite);
471  }
472  }
