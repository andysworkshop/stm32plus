89  template<class TAccessMode>
90  inline void SSD1289Orientation<LANDSCAPE,TAccessMode>::moveTo(const Rectangle& rc) const {
91 
92  this->_accessMode.writeCommand(ssd1289::HORIZONTAL_POSITION,((rc.Y+rc.Height-1) << 8) | rc.Y);
93  this->_accessMode.writeCommand(ssd1289::SET_GDDRAM_X,rc.Y);
94 
95  this->_accessMode.writeCommand(ssd1289::VERTICAL_POSITION_START,rc.X);
96  this->_accessMode.writeCommand(ssd1289::VERTICAL_POSITION_END,rc.X+rc.Width-1);
97  this->_accessMode.writeCommand(ssd1289::SET_GDDRAM_Y,rc.X);
98  }
