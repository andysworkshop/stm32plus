80  inline void LowPowerModes::doStop(uint32_t regulatorMode,uint8_t entry) {
81 
82  bool systickInterruptEnabled;
83 
84  // Systick interrupt cannot be enabled during stop mode
85  // See: STM32F103xC/D/E errata: "Debugging Stop mode and system tick timer"
86 
87  systickInterruptEnabled=(SysTick->CTRL & SysTick_CTRL_TICKINT_Msk) != 0;
88  if(systickInterruptEnabled)
89  SysTick->CTRL&=~SysTick_CTRL_TICKINT_Msk;
90 
91  // enter stop mode
92 
93  PWR_EnterSTOPMode(regulatorMode,entry);
94 
95  // re-enable clocks - you must supply this function in your app, and you must have done in order
96  // to run anything at all though you may have called it something different.
97 
98  ::SystemInit();
99 
100  // re-enable Systick if it was previously on. If a Systick would have fired during the stop mode
101  // then that tick is lost, i.e. the interrupt pending bit is not set on wakeup.
102 
103  if(systickInterruptEnabled)
104  SysTick->CTRL|=SysTick_CTRL_TICKINT_Msk;
105  }
