48  template<class TPeripheralInfo,uint16_t TTimerEvent,uint32_t TPriority,uint32_t TDmaMode>
49  inline void PwmFadeTimerDmaFeature<TPeripheralInfo,TTimerEvent,TPriority,TDmaMode>::beginFadeByTimer(Timer& timer,const uint8_t *percents,uint16_t numPercents) {
50 
51  uint16_t i;
52  uint32_t period;
53 
54  // get the timer period from the base class
55 
56  period=timer.getPeriod()+1;
57 
58  // allocate space for the percentages
59 
60  _compareValues.reset(new uint16_t[numPercents]);
61  _numCompareValues=numPercents;
62 
63  // convert each percent to a compare value
64 
65  for(i=0;i<numPercents;i++) {
66 
67  // watch out for overflow
68 
69  if(period<0xFFFFFFFF/100)
70  _compareValues[i]=static_cast<uint16_t>((period*static_cast<uint32_t>(percents[i]))/100L);
71  else
72  _compareValues[i]=(period/100L)*static_cast<uint32_t>(percents[i]);
73  }
74 
75  // set the DMA off
76 
77  this->beginWriteByTimer(timer,_compareValues.get(),numPercents);
78  }
