216  inline void Gpio::getPortAndPinSource(uint8_t& portSource,uint8_t& pinSource) const {
217 
218  switch(reinterpret_cast<uint32_t>(_peripheralAddress)) {
219  case GPIOA_BASE: portSource=GPIO_PortSourceGPIOA; break;
220  case GPIOB_BASE: portSource=GPIO_PortSourceGPIOB; break;
221  case GPIOC_BASE: portSource=GPIO_PortSourceGPIOC; break;
222  case GPIOD_BASE: portSource=GPIO_PortSourceGPIOD; break;
223  case GPIOE_BASE: portSource=GPIO_PortSourceGPIOE; break;
224  case GPIOF_BASE: portSource=GPIO_PortSourceGPIOF; break;
225  case GPIOG_BASE: portSource=GPIO_PortSourceGPIOG; break;
226  default: portSource=0; break;
227  }
228 
229  switch(_selectedPin) {
230  case GPIO_Pin_0: pinSource=GPIO_PinSource0; break;
231  case GPIO_Pin_1: pinSource=GPIO_PinSource1; break;
232  case GPIO_Pin_2: pinSource=GPIO_PinSource2; break;
233  case GPIO_Pin_3: pinSource=GPIO_PinSource3; break;
234  case GPIO_Pin_4: pinSource=GPIO_PinSource4; break;
235  case GPIO_Pin_5: pinSource=GPIO_PinSource5; break;
236  case GPIO_Pin_6: pinSource=GPIO_PinSource6; break;
237  case GPIO_Pin_7: pinSource=GPIO_PinSource7; break;
238  case GPIO_Pin_8: pinSource=GPIO_PinSource8; break;
239  case GPIO_Pin_9: pinSource=GPIO_PinSource9; break;
240  case GPIO_Pin_10: pinSource=GPIO_PinSource10; break;
241  case GPIO_Pin_11: pinSource=GPIO_PinSource11; break;
242  case GPIO_Pin_12: pinSource=GPIO_PinSource12; break;
243  case GPIO_Pin_13: pinSource=GPIO_PinSource13; break;
244  case GPIO_Pin_14: pinSource=GPIO_PinSource14; break;
245  case GPIO_Pin_15: pinSource=GPIO_PinSource15; break;
246  default: pinSource=0; break;
247  }
