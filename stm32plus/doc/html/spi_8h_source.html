164  inline bool Spi::send(const uint8_t *dataToSend,uint32_t numBytes,uint8_t *dataReceived) const {
165 
166  // wait for ready to send
167 
168  while(numBytes--) {
169 
170  while(!readyToSend())
171  if(hasError())
172  return false;
173 
174  // send the byte
175 
176  SPI_I2S_SendData(_peripheralAddress,*dataToSend++);
177 
178  if(_direction==SPI_Direction_2Lines_FullDuplex) {
179 
180  // in duplex mode and we want data, wait for it to come
181 
182  while(SPI_I2S_GetFlagStatus(_peripheralAddress,SPI_I2S_FLAG_RXNE)==RESET)
183  if(hasError())
184  return false;
185 
186  // read the byte to clear RXNE and save/discard
187 
188  if(dataReceived!=nullptr)
189  *dataReceived++=SPI_I2S_ReceiveData(_peripheralAddress);
190  else
191  SPI_I2S_ReceiveData(_peripheralAddress);
192  }
193  }
194 
195  return true;
196  }
