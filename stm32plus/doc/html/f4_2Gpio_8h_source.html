222  inline void Gpio::getPortAndPinSource(uint8_t& portSource,uint8_t& pinSource) const {
223 
224  switch(reinterpret_cast<uint32_t>(_peripheralAddress)) {
225  case GPIOA_BASE: portSource=EXTI_PortSourceGPIOA; break;
226  case GPIOB_BASE: portSource=EXTI_PortSourceGPIOB; break;
227  case GPIOC_BASE: portSource=EXTI_PortSourceGPIOC; break;
228  case GPIOD_BASE: portSource=EXTI_PortSourceGPIOD; break;
229  case GPIOE_BASE: portSource=EXTI_PortSourceGPIOE; break;
230  case GPIOF_BASE: portSource=EXTI_PortSourceGPIOF; break;
231  case GPIOG_BASE: portSource=EXTI_PortSourceGPIOG; break;
232  case GPIOH_BASE: portSource=EXTI_PortSourceGPIOH; break;
233  case GPIOI_BASE: portSource=EXTI_PortSourceGPIOI; break;
234  default: portSource=0; break;
235  }
236 
237  switch(_selectedPin) {
238  case GPIO_Pin_0: pinSource=EXTI_PinSource0; break;
239  case GPIO_Pin_1: pinSource=EXTI_PinSource1; break;
240  case GPIO_Pin_2: pinSource=EXTI_PinSource2; break;
241  case GPIO_Pin_3: pinSource=EXTI_PinSource3; break;
242  case GPIO_Pin_4: pinSource=EXTI_PinSource4; break;
243  case GPIO_Pin_5: pinSource=EXTI_PinSource5; break;
244  case GPIO_Pin_6: pinSource=EXTI_PinSource6; break;
245  case GPIO_Pin_7: pinSource=EXTI_PinSource7; break;
246  case GPIO_Pin_8: pinSource=EXTI_PinSource8; break;
247  case GPIO_Pin_9: pinSource=EXTI_PinSource9; break;
248  case GPIO_Pin_10: pinSource=EXTI_PinSource10; break;

299  inline void Gpio::initialiseAF(uint16_t pinIds,
300  uint8_t afSelection,
301  GPIOSpeed_TypeDef speed,
302  Gpio::GpioOutputType outputType,
303  Gpio::GpioPullUpDownType pullUpDownType) {
304  uint8_t source;
305  uint32_t bit;
306  GPIO_InitTypeDef init;
307 
308  (void)pullUpDownType; // unused
309 
310  _pinIds=pinIds;
311 
312  init.GPIO_Speed=speed;
313  init.GPIO_Mode=GPIO_Mode_AF;
314  init.GPIO_OType=outputType==Gpio::PUSH_PULL ? GPIO_OType_PP : GPIO_OType_OD;
315  init.GPIO_PuPd=pullUpDownType==PUPD_NONE ? GPIO_PuPd_NOPULL : (pullUpDownType==PUPD_UP ? GPIO_PuPd_UP : GPIO_PuPd_DOWN);
316  init.GPIO_Pin=pinIds;
317 
318  GPIO_Init(_peripheralAddress,&init);
319 
320  // need to configure each of the selected pins for AF
321 
322  for(bit=1,source=0;bit<=0x8000;bit<<=1,source++) {
323  if((pinIds & bit)!=0) {
324  GPIO_PinAFConfig(_peripheralAddress,source,afSelection);
325  _portBase.setPinHandler(source,this);
326  }
327  }
328  }
