78  template<Orientation TOrientation,ColourDepth TColourDepth,class TAccessMode,class TPanelTraits>
79  inline void LDS285<TOrientation,TColourDepth,TAccessMode,TPanelTraits>::initialise() {
80 
81  // reset the device
82 
83  this->_accessMode.reset();
84 
85  // start up the display
86 
87  this->_accessMode.writeCommand(lds285::SLEEP_OUT);
88  MillisecondTimer::delay(10);
89  this->_accessMode.writeCommand(lds285::DISPLAY_INVERSION_OFF);
90  this->_accessMode.writeCommand(lds285::IDLE_MODE_OFF);
91  this->_accessMode.writeCommand(lds285::NORMAL_DISPLAY_MODE_ON);
92 
93  // reset scroll area and position
94 
95  this->setScrollArea(0,320);
96  this->setScrollPosition(0);
97 
98  // memory access control comes from the orientation specialisation
99 
100  this->_accessMode.writeCommand(lds285::MEMORY_ACCESS_CONTROL,this->getMemoryAccessControl());
101 
102  // interface pixel format comes from the colour specialisation
103 
104  this->_accessMode.writeCommand(lds285::INTERFACE_PIXEL_FORMAT,this->getInterfacePixelFormat());
105 
106  // brightness enabled for GUI and to zero to start with
107 
108  setBacklight(0);
109  this->_accessMode.writeCommand(lds285::WRITE_CTRL_DISPLAY,0x24);
110  setContentAdaptiveImageType(CONTENT_ADAPTIVE_BRIGHTNESS_TYPE_GUI);
111 
112  // wait the required number of ms. before we can turn the display on
113 
114  MillisecondTimer::delay(125);
115  this->_accessMode.writeCommand(lds285::DISPLAY_ON);
116  }

126  inline void LDS285<TOrientation,TColourDepth,TAccessMode,TPanelTraits>::fadeBacklightTo(uint16_t newPercentage,int msPerStep) {
127 
128  int8_t direction;
129 
130  if(newPercentage==_backlightPercentage)
131  return;
132 
133  direction=newPercentage>_backlightPercentage ? 1 : -1;
134 
135  while(newPercentage!=_backlightPercentage) {
136  setBacklight(_backlightPercentage+direction);
137  MillisecondTimer::delay(msPerStep);
138  }
139  }
